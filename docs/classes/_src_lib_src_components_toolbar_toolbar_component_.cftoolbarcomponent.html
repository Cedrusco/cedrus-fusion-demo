Overrides CfCoreComponent.properties

Example of model:

{
    type: 'info',
    info: {
        show: true,
        icon: this.myInfoIcon,
        order: 0
    },
    content: {
        title: 'This is my toolbar title',
     template: this.myToolbarTemplate,
        order: 1
    },
    actions: {
        show: true,
        menu: this.myToobarMenu,
        order: 2
    },
    help: {
        show: true,
        icon: this.myHelpIcon,
        order: 3
    },
    close: {
        show: true,
        icon: this.myCloseIcon,
        order: 4
    },
    maximization: {
        show: true,
        maximized: false,
        maximizeIcon: this.myMaxIcon,
        minimizeIcon: this.myMinIcon,
        order: 5
    },
    expandable: {
        show: true,
        expanded: true,
        icon: this.myexpandableIcon
        order: 6
    }
}
